_Every day the snail crawls 7 feet up and every night it goes down 2 feet. How many days will it take the snail to climb to the specified height?_

```
function main() {
    let depth = parseInt(readLine(), 10);
    let days = 0;
    for (distance = 0; ; ) {
     distance += 7;
     days++;
     if (distance >= depth) {
      break;
     } else {
      distance -= 2;
     }
    }
    console.log(days);
}
```
